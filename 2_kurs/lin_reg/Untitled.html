def make_bmi(height_inch, weight_pound):
    METER_TO_INCH, KILO_TO_POUND = 39.37, 2.20462
    return (weight_pound / KILO_TO_POUND) / \
           (height_inch / METER_TO_INCH) ** 2

def error(w0, w1):
    error = 0
    v , s = data.shape
    for n in range(v):
        error += (data['Height'].iloc[n] - (w0+w1*data['Weight'].iloc[n]))**2
    return error
